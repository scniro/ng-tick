engine

  • inject and instantiate a new engine
  • specify an interval
  • be sure to remove the timer from memory via .stop() 
module.controller('ctrl', ['$scope', 'engine', function ($scope, engine) {
    var eng = new engine();

    eng.start(1000).on(function (interval) {
        // do stuff
    });

    $scope.$on('$destroy', function () {
        eng.stop(); // clear from memory
    });
}]);
