module.controller('ctrl', ['$scope', '$timeout', function ($scope, $timeout) {
    
    // immediate invocation requires a $timeout wrapper
    $timeout(function() {
        $scope.myclock.start();
    });
    
    // optional event callback if handle specified
    $scope.$on('myclock:tick', function(event, status) {
        console.log(status);
    })
}]);
